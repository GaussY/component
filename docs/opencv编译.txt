1.opencv编译
cmake -DCMAKE_BUILD_TYPE=Release -DCMAKE_INSTALL_PREFIX=/home/yyy/third_party/opencv -DBUILD_SHARED_LIBS=ON -DBUILD_STATIC_LIBS=OFF -DBUILD_EXAMPLES=OFF -DBUILD_TESTS=OFF -DBUILD_PERF_TESTS=OFF -DWITH_GTK=OFF -DWITH_GTK_2_X=OFF -DWITH_QT=OFF -DWITH_OPENGL=OFF -DWITH_DC1394=OFF -DWITH_FFMPEG=ON -DWITH_V4L=OFF -DCMAKE_CXX_STANDARD=11 -DCMAKE_CXX_STANDARD_REQUIRED=ON -DCMAKE_CXX_EXTENSIONS=ON -DCMAKE_CXX_FLAGS="-std=gnu++11 -D_GLIBCXX_USE_CXX11_ABI=0 -w -Wno-error -Wno-address -Wno-error=address -Wno-pointer-arith" -DCMAKE_C_FLAGS="-w -Wno-error -Wno-address -Wno-error=address -Wno-pointer-arith" -DCMAKE_C_COMPILER=${CC} -DCMAKE_CXX_COMPILER=${CXX} -DCMAKE_CXX_FLAGS_RELEASE="-O2 -DNDEBUG" -DCMAKE_BUILD_WITH_INSTALL_RPATH=ON ..
